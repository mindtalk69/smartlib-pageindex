    3. Iterates through each key-value pair in settings
    4. Validates special settings:

    Response format:
    ```json

      "success": true,
      "message": "Updated 3 setting(s).",
      "updated_keys": ["max_active_users", "app_name", "primary_color"]
    }
    ```

    Error handling:
    - 400 for max_active_users validation failures:
      - Not a valid integer
      - Less than 1
      - Less than current active user count

    Implementation pattern:
    ```python
    from modules.database import count_active_users, set_app_setting

    @app.post("/api/v1/admin/settings/update", response_model=SettingsUpdateResponse)
    def update_app_settings(
        settings_data: SettingsUpdateRequest,
        db: Session = Depends(get_db),
        current_user: User = Depends(get_current_admin_user),
    ):
        """Update application settings (admin only)."""
        updated_keys = []
        current_active = count_active_users()

        for key, value in settings_data.settings.items():
            # Special validation for max_active_users
            if key == 'max_active_users':
                try:
                    max_users = int(value)
                except (ValueError, TypeError):
                    raise HTTPException(
                        status_code=status.HTTP_400_BAD_REQUEST,
                        detail="max_active_users must be a valid integer."
                    )

                if max_users < 1:
                    raise HTTPException(
                        status_code=status.HTTP_400_BAD_REQUEST,
                        detail="max_active_users must be at least 1."
                    )

                if max_users < current_active:
                    raise HTTPException(
                        status_code=status.HTTP_400_BAD_REQUEST,
                        detail=f"Cannot set limit to {max_users} - currently have {current_active} active users."
                    )

            # Update or insert setting
            existing = db.get(AppSettings, key)
            if existing:
                existing.value = str(value)
            else:
                new_setting = AppSettings(key=key, value=str(value))
                db.add(new_setting)

            updated_keys.append(key)

        db.commit()

        return {
            "success": True,
            "message": f"Updated {len(updated_keys)} setting(s).",
            "updated_keys": updated_keys,
        }
    ```
