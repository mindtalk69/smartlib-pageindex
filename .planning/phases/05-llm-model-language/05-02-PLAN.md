    **1. Define HealthStatus interface:**
    ```tsx
    interface HealthStatus {
      status: 'healthy' | 'degraded' | 'offline' | 'unknown'
      lastHealthCheck: string | null
      errorMessage: string | null
      responseTime?: number
    }
    ```

    **2. Hook functionality:**
    - Track health status per provider
    - Provide checkHealth function to trigger health check
    - Auto-refresh health status on interval (optional)
    - Return health status for specific provider

    **3. Implementation:**
    ```tsx
    import { useState, useCallback } from 'react'
    import { api } from '@/lib/apiClient'

    export function useProviderHealth() {
      const [healthStatuses, setHealthStatuses] = useState<Record<number, HealthStatus>>({})
      const [checkingHealth, setCheckingHealth] = useState<Set<number>>(new Set())

      const checkHealth = useCallback(async (providerId: number) => {
        setCheckingHealth(prev => new Set(prev).add(providerId))
        try {
          const result = await api.post(`/api/v1/admin/providers/${providerId}/test`)
          setHealthStatuses(prev => ({
            ...prev,
            [providerId]: {
              status: result.status,
              lastHealthCheck: new Date().toISOString(),
              errorMessage: result.error || null,
              responseTime: result.response_time
            }
          }))
          return result
        } catch (error) {
          setHealthStatuses(prev => ({
            ...prev,
            [providerId]: {
              status: 'offline',
              lastHealthCheck: new Date().toISOString(),
              errorMessage: error instanceof Error ? error.message : 'Health check failed'
            }
          }))
          throw error
        } finally {
          setCheckingHealth(prev => {
            const next = new Set(prev)
            next.delete(providerId)
            return next
          })
        }
      }, [])

      const getHealthStatus = (providerId: number): HealthStatus | undefined => {
        return healthStatuses[providerId]
      }

      const isChecking = (providerId: number): boolean => {
        return checkingHealth.has(providerId)
      }

      return {
        healthStatuses,
        checkHealth,
        getHealthStatus,
        isChecking
      }
    }
    ```
