  <action>
    Create custom hook for language management:

    **1. Define LlmLanguage interface:**
    ```tsx
    interface LlmLanguage {
      id: number
      language_code: string  // e.g., 'en', 'zh', 'es'
      language_name: string  // e.g., 'English', 'Chinese', 'Spanish'
      is_active: boolean
      created_by: string | null
      created_at: string
    }
    ```

    **2. Hook functionality:**
    - Fetch languages from GET /api/v1/admin/languages on mount
    - Provide CRUD functions: addLanguage, updateLanguage, deleteLanguage
    - Provide toggle function: toggleActive
    - Track loading and error states
    - Return languages list sorted by language_name

    **3. Implementation:**
    ```tsx
    import { useState, useEffect, useCallback } from 'react'
    import { api } from '@/lib/apiClient'

    export function useLanguages() {
      const [languages, setLanguages] = useState<LlmLanguage[]>([])
      const [isLoading, setIsLoading] = useState(true)
      const [error, setError] = useState<string | null>(null)

      const fetchLanguages = useCallback(async () => {
        setIsLoading(true)
        setError(null)
        try {
          const data = await api.get('/api/v1/admin/languages')
          setLanguages(data.languages || [])
        } catch (err) {
          setError(err instanceof Error ? err.message : 'Failed to fetch languages')
        } finally {
          setIsLoading(false)
        }
      }, [])

      const addLanguage = async (language: Omit<LlmLanguage, 'id' | 'created_at' | 'created_by'>) => {
        const result = await api.post('/api/v1/admin/languages/add', language)
        await fetchLanguages()
        return result
      }

      const updateLanguage = async (id: number, updates: Partial<LlmLanguage>) => {
        const result = await api.post(`/api/v1/admin/languages/edit/${id}`, updates)
        await fetchLanguages()
        return result
      }

      const deleteLanguage = async (id: number) => {
        const result = await api.delete(`/api/v1/admin/languages/delete/${id}`)
        await fetchLanguages()
        return result
      }

      const toggleActive = async (id: number, currentStatus: boolean) => {
        const result = await updateLanguage(id, { is_active: !currentStatus })
        return result
      }

      useEffect(() => {
        fetchLanguages()
      }, [fetchLanguages])

      return {
        languages,
        isLoading,
        error,
        refreshLanguages: fetchLanguages,
        addLanguage,
        updateLanguage,
        deleteLanguage,
        toggleActive
      }
    }
    ```

    **4. Export the hook**
