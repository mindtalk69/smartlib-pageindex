    **1. Define Provider interface:**
    ```tsx
    interface LLMProvider {
      id: number
      name: string
      provider_type: string  // 'azure_openai', 'ollama', 'openai'
      base_url: string | null
      api_key?: string  // Only for send, not received
      is_active: boolean
      is_default: boolean
      priority: number
      config: Record<string, unknown>
      last_health_check: string | null
      health_status: 'healthy' | 'degraded' | 'offline' | null
      error_message: string | null
      created_at: string
      updated_at: string
    }
    ```

    **2. Hook functionality:**
    - Fetch providers from GET /api/v1/admin/providers on mount
    - Provide CRUD functions: addProvider, updateProvider, deleteProvider
    - Provide action functions: testConnection, discoverModels, updatePriorities
    - Track loading and error states
    - Return providers list sorted by priority

    **3. Implementation:**
    ```tsx
    import { useState, useEffect, useCallback } from 'react'
    import { api } from '@/lib/apiClient'

    export function useProviders() {
      const [providers, setProviders] = useState<LLMProvider[]>([])
      const [isLoading, setIsLoading] = useState(true)
      const [error, setError] = useState<string | null>(null)

      const fetchProviders = useCallback(async () => {
        setIsLoading(true)
        setError(null)
        try {
          const data = await api.get('/api/v1/admin/providers')
          setProviders(data.providers || [])
        } catch (err) {
          setError(err instanceof Error ? err.message : 'Failed to fetch providers')
        } finally {
          setIsLoading(false)
        }
      }, [])

      const addProvider = async (provider: Omit<LLMProvider, 'id' | 'created_at' | 'updated_at'>) => {
        const result = await api.post('/api/v1/admin/providers', provider)
        await fetchProviders()
        return result
      }

      const updateProvider = async (id: number, updates: Partial<LLMProvider>) => {
        const result = await api.put(`/api/v1/admin/providers/${id}`, updates)
        await fetchProviders()
        return result
      }

      const deleteProvider = async (id: number) => {
        const result = await api.delete(`/api/v1/admin/providers/${id}`)
        await fetchProviders()
        return result
      }

      const testConnection = async (id: number) => {
        return await api.post(`/api/v1/admin/providers/${id}/test`)
      }

      const discoverModels = async (id: number) => {
        return await api.post(`/api/v1/admin/providers/${id}/discover-models`)
      }

      const updatePriorities = async (priorities: Array<{id: number; priority: number}>) => {
        return await api.post('/api/v1/admin/providers/priority', { priorities })
      }

      useEffect(() => {
        fetchProviders()
      }, [fetchProviders])

      return {
        providers,
        isLoading,
        error,
        refreshProviders: fetchProviders,
        addProvider,
        updateProvider,
        deleteProvider,
        testConnection,
        discoverModels,
        updatePriorities
      }
    }
    ```

    **4. Export the hook** for use in Providers page
