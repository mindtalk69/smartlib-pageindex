      temperature: number | null
      streaming: boolean
      description: string | null

    **2. Hook functionality:**
    - Fetch models from GET /api/v1/admin/models on mount
    - Provide CRUD functions: addModel, updateModel, deleteModel
    - Provide action functions: setDefault, setMultimodal, validateDeployment
    - Track loading and error states
    - Return models list

    **3. Implementation:**
    ```tsx
    import { useState, useEffect, useCallback } from 'react'
    import { api } from '@/lib/apiClient'

    export function useModels() {
      const [models, setModels] = useState<ModelConfig[]>([])
      const [isLoading, setIsLoading] = useState(true)
      const [error, setError] = useState<string | null>(null)

      const fetchModels = useCallback(async () => {
        setIsLoading(true)
        setError(null)
        try {
          const data = await api.get('/api/v1/admin/models')
          setModels(data.models || [])
        } catch (err) {
          setError(err instanceof Error ? err.message : 'Failed to fetch models')
        } finally {
          setIsLoading(false)
        }
      }, [])

      const addModel = async (model: Omit<ModelConfig, 'id' | 'created_at' | 'provider_obj'>) => {
        const result = await api.post('/api/v1/admin/models/add', model)
        await fetchModels()
        return result
      }

      const updateModel = async (id: number, updates: Partial<ModelConfig>) => {
        const result = await api.post(`/api/v1/admin/models/edit/${id}`, updates)
        await fetchModels()
        return result
      }

      const deleteModel = async (id: number) => {
        const result = await api.delete(`/api/v1/admin/models/delete/${id}`)
        await fetchModels()
        return result
      }

      const setDefault = async (id: number) => {
        return await api.post(`/api/v1/admin/models/set-default/${id}`)
      }

      const setMultimodal = async (id: number) => {
        return await api.post(`/api/v1/admin/models/set-multimodal/${id}`)
      }

      const validateDeployment = async (config: {
        deployment_name: string
        temperature: number | null
        streaming: boolean
        provider_id?: number
      }) => {
        return await api.post('/api/v1/admin/models/validate', config)
      }

      useEffect(() => {
        fetchModels()
      }, [fetchModels])

      return {
        models,
        isLoading,
        error,
        refreshModels: fetchModels,
        addModel,
        updateModel,
        deleteModel,
        setDefault,
        setMultimodal,
        validateDeployment
      }
    }
    ```

    **4. Export the hook**
