    Set up React Router v7 routing in App.tsx:

    **1. Import Routes and Route from react-router-dom:**
    ```tsx
    import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom'
    ```

    **2. Create placeholder components:**
    - AdminDashboard: Returns "Admin Dashboard" with h1
    - AdminLogin: Returns "Admin Login" with h1 (temporary, will use shared login)

    **3. Configure routes:**
    - / → AdminDashboard
    - /login → AdminLogin
    - /* → Navigate to / (catch-all for SPA)

    **4. Wrap Routes in BrowserRouter:**
    ```tsx
    export function App() {
      return (
        <BrowserRouter>
          <Routes>
            <Route path="/" element={<AdminDashboard />} />
            <Route path="/login" element={<AdminLogin />} />
            <Route path="*" element={<Navigate to="/" replace />} />
          </Routes>
        </BrowserRouter>
      )
    }
    ```

    **5. Export App as default**

    Note: This is a temporary structure. Authentication and protected routes will be added in plan 03-03.
