    Create AdminAuthContext for admin authentication state:

    **1. Context definition:**
    ```tsx
    interface AdminUser {
      id: string
      username: string
      email?: string
      is_admin: boolean
    }

    interface AdminAuthContextType {
      admin: AdminUser | null
      isLoading: boolean
      isAuthenticated: boolean
      isAdmin: boolean
      logout: () => Promise<void>
      checkAuth: () => Promise<void>
    }
    ```

    **2. AdminAuthProvider implementation:**
    - On mount: Call /api/me to validate JWT token
    - Extract user data from response
    - Check is_admin claim - if false, redirect to /app with access denied
    - Store admin user in state
    - Handle loading state during validation

    **3. checkAuth function:**
    ```tsx
    const checkAuth = async () => {
      setIsLoading(true)
      try {
        const token = localStorage.getItem('auth_token')
        if (!token) {
          // Redirect to login
          window.location.href = '/app/login'
          return
        }

        const response = await fetch('/api/me', {
          headers: { 'Authorization': `Bearer ${token}` }
        })

        if (!response.ok) {
          if (response.status === 401) {
            // Token invalid - redirect to login
            localStorage.removeItem('auth_token')
            window.location.href = '/app/login'
            return
          }
          throw new Error('Auth check failed')
        }

        const data = await response.json()
        if (!data.user?.is_admin) {
          // Not admin - redirect to main app
          window.location.href = '/app?error=admin_required'
          return
        }

        setAdmin(data.user)
      } catch (err) {
        console.error('Auth check failed:', err)
        window.location.href = '/app/login'
      } finally {
        setIsLoading(false)
      }
    }
    ```

    **4. logout function:**
    - Call POST /api/v1/auth/logout
    - Clear token from localStorage
    - Clear admin state
    - Redirect to /app/login

    **5. Effect:**
    - Run checkAuth on mount
    - No interval/polling needed

    **6. Export hook:**
    ```tsx
    export function useAdminAuth() {
      const context = useContext(AdminAuthContext)
      if (!context) {
        throw new Error('useAdminAuth must be used within AdminAuthProvider')
      }
      return context
    }
    ```

    }

    **2. Implementation:**
    ```tsx
    export function ProtectedRoute({ children }: ProtectedRouteProps) {
      const { admin, isLoading, isAuthenticated, isAdmin } = useAdminAuth()
      const navigate = useNavigate()

      useEffect(() => {
        if (!isLoading) {
          if (!isAuthenticated) {
            // Not logged in - redirect to main app login
            window.location.href = '/app/login'
          } else if (!isAdmin) {
            // Not admin - redirect to main app with error
            window.location.href = '/app?error=admin_required'
          }
        }
      }, [isLoading, isAuthenticated, isAdmin, navigate])

      if (isLoading) {
        return (
          <div className="flex items-center justify-center min-h-screen">
            <div className="text-muted-foreground">Loading...</div>
          </div>
        )
      }

      if (!isAuthenticated || !admin?.is_admin) {
        return null // Redirect will happen via useEffect
      }

      return <>{children}</>
    }
    ```

    **3. Loading state:**
    - Show centered "Loading..." text while validating auth
    - Use muted-foreground color for subtle appearance

    **4. Error handling:**
    - useEffect handles redirects for unauthorized access
    - Return null during redirect to prevent flash of content

      <ProtectedRoute>
        <AdminLayout>

        </AdminLayout>
      </ProtectedRoute>
