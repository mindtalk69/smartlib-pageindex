
    **2. Add action handlers in UserDialog:**
    ```tsx
    const handleToggleAdmin = async (userId: string) => {
      try {
        await actions.toggleAdmin(userId)
        onSuccess('Admin status updated')
        onRefresh() // Refresh parent list
      } catch (err) {
        onError('Failed to update admin status')
      }
    }

    const handleResetPassword = async (userId: string) => {
      try {
        const result = await actions.resetPassword(userId)
        onSuccess(`Password reset. Temp password: ${result.tempPassword}`)
        // Copy to clipboard
        navigator.clipboard.writeText(result.tempPassword)
      } catch (err) {
        onError('Failed to reset password')
      }
    }
    ```

    **3. Add UserActions to dialog footer:**
    ```tsx
    <DialogFooter className="flex-col gap-2 border-t pt-4">
    ```
